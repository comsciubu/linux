5. การเขียนคำสั่ง Bash¶

การใช้คำสั่งภาษา Bash

5.1. Introduction¶

  1. Open terminal : gnome- terminal, konsole
  2. What’s this? mkdir -p 1104154/scripting; cd 1104154/scripting
  3. Two ways to say hello

3.1. interactive (live) scripting: type command at the prompt

echo ‘hello world’

3.2. non-interactive:

  • create file .sh
vi hello.sh

gedit hello.sh
  • put command in the file
  • save
  • execute it
chmod +x hello.sh

./hello.sh
  1. What’s your shell?
  2. Pick your editor : vi, nano
  3. Fast fingers

5.2. Variables¶

  1. Refers to data in memory: VARNAME=value
age=29
name='Paul Phoenix'
name="Angelina Jolie"
  1. Access variable value: $VARNAME or ${VARNAME}
echo $age
echo $name
  1. Reading value from keyboard: read VARNAME
read age
echo $age
  1. Ready to use variables on your shell: printenv
  2. Arithmetic operations
z=`echo $x + $y | bc`

# +,   -,   ++,   --,   *,   /,   %,  ^
# !,  &&, ||,   >,   >=,   <,   <=,  ==, !=

z=`expr $x + $y`

# |,  &,  <,  <=,  =,  !=,  >=,  >,  +,  -,  \*,  /,  %

z=$(( x + y ))

# +,   -,   ++,   --,   *,   /,   %,  **

5.3. If/Else¶

  1. One-way
if [ $x -eq $y ]; then
  echo "x == y"
fi
  1. Two-way
if [ $x -le $y ]; then
  echo "x <= y"
else
  echo "x > y"
fi
  1. Multi-way
if [ $x -eq $y ]; then
  echo "x == y"
elif (( x > 20 )); then
  echo "x > 20"
elif [ -f /bin/sh ] && [ -x /bin/sh ]; then
  echo "you’ve got shell"
else
  echo 'hmmmmmmmm'
  echo 'what do I do now?'
fi
  1. Operators
  • Condition Expression with binary operators
[ $a  operator  $b ]

# -eq ,-ne, -lt, -le, -gt, -ge
  • Condition Expression with unary operators
[ operator $a ]

# -s, -f, -d, -x, -w, -r
  • Logical operators
ConditionExpression operator ConditionExpression

# &&, ||

5.6. while¶

while condition
do
  statements
  statements
done

i=0
while [ $i -lt 10 ]
do
  i=`expr $i + 1`
  echo “i = $i”
done

i=0
while((i < 10))
do
  ((i++))
  echo “i = $i”
done

5.7. Cases¶

case $var in
  a) statements ;;
  b) statements ;;

  *) statments ;;
esac

read grade
case $grade in
4) echo "best" ;;
3) echo "good" ;;
2) echo "okay" ;;
1) echo "try harder" ;;
0) echo "fail" ;;
*) echo "wrong input" ;;
esac

5.8. Functions¶

functionname()
{
statements
  statements
}

functionname
functionname 1 2 3 ...

fac()
{
  if [ "$1" -gt 1 ]
     then
    n=`expr $1 - 1`
    r=`fac $n`
    result=`expr $1 \* $r`
    echo $result
  else
    echo 1
  fi
}

echo "Enter a number: "
read n
echo "$n! = `fac $n`"

5.9. Files¶

show()
{
case $1 in

    -c) cat $1 ;;
    -g) grep -e $2 $1 ;;
    -h) head $1 ;;
    -s) sed -i '/$2/$3' $1 ;;
esac
}

show -c users.txt
show -g users.txt Paul
show -s users.txt Paul Nina

5.11. Advanced¶

  1. Bash Array
a=(10 30 20 30 50 40)
u=( ${a[@]} )
echo ${a[0]}            # 10
echo ${a[*]}            # 10 30 20 30 50 40
echo ${a[@]}            # 10 30 20 30 50 40
echo ${#a[@]}           # 6
echo ${#a}              # 2   #number of characters in a[0]
echo ${a[@]:2:3}        # 20 30 50
echo ${a[@]/30/15}      # 10 15 20 15 50 40
  1. Bash for
a=(10 30 20 30 50 40)
for (( i=0; i<5; i++ ))
do
  printf "%d = %d\n" $i ${a[$i]}
done
  1. Bash string
${#string}            # length of $string

${string:pos}         # extract substring from $string at $pos
${string:pos:length}  # extract substring from $pos to $pos+$length

${string#term}        # strip shortest $term from front
${string##term}       # strip longest $term from front
${string%term}        # strip shortest $term from back
${string%%term}       # strip longest $term from back

${string/old/new}     # replace first match of $old with $new
${string//old/new}    # replace all matches of $old with $new
${string/#old/new}    # if $old matches front of $string replace it
${string/%old/new}    # if $old matches end of $string replace it
